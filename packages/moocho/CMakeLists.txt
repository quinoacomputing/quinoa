
INCLUDE(TribitsPackageMacros)
INCLUDE(TribitsAddOptionAndDefine)

TRIBITS_PACKAGE(MOOCHO)

ASSERT_DEFINED(${PROJECT_NAME}_ENABLE_Fortran)
TRIBITS_ADD_OPTION_AND_DEFINE(${PACKAGE_NAME}_ENABLE_QPKWIK
  HAVE_MOOCHO_QPKWIK
  "Enable support for QPKWIK"
  ${${PROJECT_NAME}_ENABLE_Fortran}
  )

ADD_SUBDIRECTORY(src)

ASSERT_DEFINED(${PACKAGE_NAME}_ENABLE_ThyraCore)
IF (${PACKAGE_NAME}_ENABLE_ThyraCore)
  ADD_SUBDIRECTORY(thyra)
ENDIF()

TRIBITS_ADD_EXAMPLE_DIRECTORIES(example)

TRIBITS_ADD_TEST_DIRECTORIES(test)

TRIBITS_EXCLUDE_AUTOTOOLS_FILES()

TRIBITS_PACKAGE_POSTPROCESS()
