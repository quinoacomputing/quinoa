INCLUDE(TribitsSubPackageMacros)
INCLUDE(TribitsAddExecutable)
INCLUDE(TribitsLibraryMacros)

TRIBITS_SUBPACKAGE(Exo2mat)

INCLUDE_DIRECTORIES(
  "${CMAKE_CURRENT_SOURCE_DIR}"
)

IF(${PACKAGE_NAME}_ENABLE_MATLAB)
  INCLUDE_DIRECTORIES(${MATLAB_INCLUDE_DIRS})
  LINK_DIRECTORIES(${MATLAB_LIBRARY_DIRS})
  SET(DEPMATLIB mx mex mat)
  ADD_DEFINITIONS(-DHAS_MATLAB)
ENDIF()

SET(CODE exo2mat)
SET(HEADERS "")

TRIBITS_ADD_EXECUTABLE(${CODE} NOEXEPREFIX NOEXESUFFIX SOURCES exo2mat.c DEPLIBS suplib exodus ${DEPMATLIB} COMM serial mpi)
install_executable(${CODE})

TRIBITS_SUBPACKAGE_POSTPROCESS()
