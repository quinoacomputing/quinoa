00066 /* bit25 of CX tells us whether AES is enabled. */
00067 R123_STATIC_INLINE int haveAESNI(){
00068     unsigned int eax, ebx, ecx, edx;
00069     __asm__ __volatile__ ("cpuid": "=a" (eax), "=b" (ebx), "=c" (ecx), "=d" (edx) :
00070                       "a" (1));

00094 /* Is there an intrinsic to assemble an __m128i from two 64-bit words? 
00095    If not, use the 4x32-bit intrisic instead.  N.B.  It looks like Intel
00096    added _mm_set_epi64x to icc version 12.1 in Jan 2012.
00097 */
00098 R123_STATIC_INLINE __m128i _mm_set_epi64x(uint64_t v1, uint64_t v0){
00099     union{
00100         uint64_t u64;
00101         uint32_t u32[2];
00102     } u1, u0;
00103     u1.u64 = v1;
00104     u0.u64 = v0;
00105     return _mm_set_epi32(u1.u32[1], u1.u32[0], u0.u32[1], u0.u32[0]);
00106 }

00108 /* _mm_extract_lo64 abstracts the task of extracting the low 64-bit
00109    word from an __m128i.  The _mm_cvtsi128_si64 intrinsic does the job
00110    on 64-bit platforms.  Unfortunately, both MSVC and Open64 fail
00111    assertions in ut_M128.cpp and ut_carray.cpp when we use the
00112    _mm_cvtsi128_si64 intrinsic.  (See
00113    https://bugs.open64.net/show_bug.cgi?id=873 for the Open64 bug).
00114    On 32-bit platforms, there's no MOVQ, so there's no intrinsic.
00115    Finally, even if the intrinsic exists, it may be spelled with or
00116    without the 'x'.
00117 */
00118 #if !defined(__x86_64__) || defined(_MSC_VER) || defined(__OPEN64__)
00119 R123_STATIC_INLINE uint64_t _mm_extract_lo64(__m128i si){
00120     union{
00121         uint64_t u64[2];
00122         __m128i m;
00123     }u;
00124     _mm_store_si128(&u.m, si);
00125     return u.u64[0];
00126 }
00127 #elif defined(__llvm__) || defined(__ICC)
00128 R123_STATIC_INLINE uint64_t _mm_extract_lo64(__m128i si){
00129     return (uint64_t)_mm_cvtsi128_si64(si);
00130 }
00131 #else /* GNUC, others */
00132 /* FWIW, gcc's emmintrin.h has had the 'x' spelling
00133    since at least gcc-3.4.4.  The no-'x' spelling showed up
00134    around 4.2. */

00181 R123_STATIC_INLINE r123m128i& operator++(r123m128i& v){
00182     __m128i& c = v.m;
00183     __m128i zeroone = _mm_set_epi64x(R123_64BIT(0), R123_64BIT(1));
00184     c = _mm_add_epi64(c, zeroone);
00185     //return c;
00186 #if R123_USE_SSE4_1
00187     __m128i zerofff = _mm_set_epi64x(0, ~(R123_64BIT(0)));
00188     if( R123_BUILTIN_EXPECT(_mm_testz_si128(c,zerofff), 0) ){
00189         __m128i onezero = _mm_set_epi64x(R123_64BIT(1), R123_64BIT(0));
00190         c = _mm_add_epi64(c, onezero);
00191     }
00192 #else
00193     unsigned mask  = _mm_movemask_ps( _mm_castsi128_ps(_mm_cmpeq_epi32(c, _mm_setzero_si128())));
00194     // The low two bits of mask are 11 iff the low 64 bits of
00195     // c are zero.
00196     if( R123_BUILTIN_EXPECT((mask&0x3) == 0x3, 0) ){
00197         __m128i onezero = _mm_set_epi64x(1,0);
00198         c = _mm_add_epi64(c, onezero);
00199     }
00200 #endif
00201     return v;
