00041 /*
00042 // Macros _Foo_tpl are code generation 'templates'  They define
00043 // inline functions with names obtained by mangling Foo and the
00044 // macro arguments.  E.g.,
00045 //   _mulhilo_tpl(32, uint32_t, uint64_t)
00046 // expands to a definition of:
00047 //   mulhilo32(uint32_t, uint32_t, uint32_t *, uint32_t *)
00048 // We then 'instantiate the template' to define
00049 // several different functions, e.g.,
00050 //   mulhilo32
00051 //   mulhilo64
00052 // These functions will be visible to user code, and may
00053 // also be used later in subsequent templates and definitions.
00054 
00055 // A template for mulhilo using a temporary of twice the word-width.
00056 // Gcc figures out that this can be reduced to a single 'mul' instruction,
00057 // despite the apparent use of double-wide variables, shifts, etc.  It's
00058 // obviously not guaranteed that all compilers will be that smart, so
00059 // other implementations might be preferable, e.g., using an intrinsic
00060 // or an asm block.  On the other hand, for 32-bit multiplies,
00061 // this *is* perfectly standard C99 - any C99 compiler should 
00062 // understand it and produce correct code.  For 64-bit multiplies,
00063 // it's only usable if the compiler recognizes that it can do
00064 // arithmetic on a 128-bit type.  That happens to be true for gcc on
00065 // x86-64, and powerpc64 but not much else.
00066 */
00067 #define _mulhilo_dword_tpl(W, Word, Dword)                              \
00068 R123_CUDA_DEVICE R123_STATIC_INLINE Word mulhilo##W(Word a, Word b, Word* hip){ \
00069     Dword product = ((Dword)a)*((Dword)b);                              \
00070     *hip = product>>W;                                                  \
00071     return (Word)product;                                               \
00072 }

00074 /*
00075 // A template for mulhilo using gnu-style asm syntax.
00076 // INSN can be "mulw", "mull" or "mulq".  
00077 // FIXME - porting to other architectures, we'll need still-more conditional
00078 // branching here.  Note that intrinsics are usually preferable.
00079 */
00080 #ifdef __powerpc__
00081 #define _mulhilo_asm_tpl(W, Word, INSN)                         \
00082 R123_STATIC_INLINE Word mulhilo##W(Word ax, Word b, Word *hip){ \
00083     Word dx = 0;                                                \
00084     __asm__("\n\t"                                              \
00085         INSN " %0,%1,%2\n\t"                                    \
00086         : "=r"(dx)                                              \
00087         : "r"(b), "r"(ax)                                       \
00088         );                                                      \
00089     *hip = dx;                                                  \
00090     return ax*b;                                                \
00091 }
00092 #else
00093 #define _mulhilo_asm_tpl(W, Word, INSN)                         \
00094 R123_STATIC_INLINE Word mulhilo##W(Word ax, Word b, Word *hip){      \
00095     Word dx;                                                    \
00096     __asm__("\n\t"                                              \
00097         INSN " %2\n\t"                                          \
00098         : "=a"(ax), "=d"(dx)                                    \
00099         : "r"(b), "0"(ax)                                       \
00100         );                                                      \
00101     *hip = dx;                                                  \
00102     return ax;                                                  \
00103 }
00104 #endif /* __powerpc__ */

00124 /*
00125 // A template for mulhilo using only word-size operations and
00126 // C99 operators (no adc, no mulhi).  It
00127 // requires four multiplies and a dozen or so shifts, adds
00128 // and tests.  It's not clear what this is good for, other than
00129 // completeness.  On 32-bit platforms, it could be used to
00130 // implement philoxNx64, but on such platforms both the philoxNx32
00131 // and the threefryNx64 cbrngs are going to have much better
00132 // performance.  It is enabled below by R123_USE_MULHILO64_C99,
00133 // but that is currently (Sep 2011) not set by any of the
00134 // features/XXfeatures.h headers.  It can, of course, be
00135 // set with a compile-time -D option.
00136 */
00137 #define _mulhilo_c99_tpl(W, Word) \
00138 R123_STATIC_INLINE Word mulhilo##W(Word a, Word b, Word *hip){ \
00139     const unsigned WHALF = W/2;                                    \
00140     const Word LOMASK = ((((Word)1)<<WHALF)-1);                    \
00141     Word lo = a*b;               /* full low multiply */           \
00142     Word ahi = a>>WHALF;                                           \
00143     Word alo = a& LOMASK;                                          \
00144     Word bhi = b>>WHALF;                                           \
00145     Word blo = b& LOMASK;                                          \
00146                                                                    \
00147     Word ahbl = ahi*blo;                                           \
00148     Word albh = alo*bhi;                                           \
00149                                                                    \
00150     Word ahbl_albh = ((ahbl&LOMASK) + (albh&LOMASK));                   \
00151     Word hi = ahi*bhi + (ahbl>>WHALF) +  (albh>>WHALF);                 \
00152     hi += ahbl_albh >> WHALF; /* carry from the sum of lo(ahbl) + lo(albh) ) */ \
00153     /* carry from the sum with alo*blo */                               \
00154     hi += ((lo >> WHALF) < (ahbl_albh&LOMASK));                         \
00155     *hip = hi;                                                          \
00156     return lo;                                                          \
00157 }
